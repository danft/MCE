def get_center(theta, a, b, h, x, y):
    
    m = tan(theta)

    B = a**2*m**2 + b**2
    A = 4*a**2*b**2*(1+m**2)
    D = 4*h**2
    c = sqrt((B*A-D*B**2)/A)
    alx = (-a**2*m*c+a*b*sqrt(a**2*m**2+b**2-c**2))/(a**2*m**2+b**2)
    aly = alx * m + c
    px = h * cos(theta)
    py = h * sin(theta)
    bx = alx - px
    by = aly - py
    xc = bx * cos(theta) + by * sin(theta)
    yc = bx * sin(theta) - by * cos(theta)

    return xc, yc

def fk(theta, a, b, h, x, y):

    #print("bla", theta, tu)
    xc, yc = get_center(theta, a, b, h, x, y)
    X = xc-x
    Y = yc-y
    return (X*cos(theta) + Y*sin(theta))**2/a**2 + (X*sin(theta) - Y*cos(theta))**2/b**2 - 1

def g(a, b, h, x, y, tu):

def find_roots(f, l, r, phase, _args, nfev=0):
    if r - l < 1e-4:
        return [], 0
    
    print(l, r, phase)
    retl = []
    
    
    if phase == 0:
        res = minimize_scalar(f, bounds=(l, r), method='bounded', tol=1e-5, args=_args)
        l1 = l
        r1 = res.x
        
        l2 = res.x
        r2 = r
        nfev += res.nfev
        
        if r1 - l1 < 1e-4 or r2 - l2 < 1e-4:
            re = find_roots(f, l, r, phase+1, _args, 0)
            retl.extend(re[0])
            nfev += re[1]
            
        else:
            re = find_roots(f, l1, r1, phase, _args, 0)
            retl.extend(re[0])
            nfev += re[1]
            re = find_roots(f, l2, r2, phase, _args, 0)
            retl.extend(re[0])
            nfev += re[1]
                    
    elif phase == 1:
        res = minimize_scalar(lambda t, a, b, h, x, y: -f(t, a, b, h, x, y), bounds=(l, r), method='bounded', tol=1e-5, args=_args)
        l1 = l
        r1 = res.x
              
        l2 = res.x
        r2 = r
        nfev += res.nfev

        
        re = find_roots(f, l1, r1, phase+1, _args, 0)
        retl.extend(re[0])
        nfev += re[1]
        re = find_roots(f, l2, r2, phase+1, _args, 0)
        retl.extend(re[0])
        nfev += re[1]
    
    else:
        
        if f(l, *_args) * f(r, *_args) > 0:
            return [], 0
        
        res = optimize.toms748(f, l, r, args=_args, full_output=True)
        nfev += res[1].iterations
        retl.append(res[1])
    
    return retl, nfev
